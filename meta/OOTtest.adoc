== Actors
[source,cypher]
----
CREATE ( Ganondorf:actor { type:'boss' }),
( Link:actor { type:'playerCharacter' }),
( Zelda:actor { type:'NPC' }),
( Sheik:actor { type:'NPC' })

CREATE
(Ganondorf)-[:KIDNAPS]->(Zelda),
(Zelda)-[:ALTER_EGO]->(Sheik),
(Sheik)-[:AIDS]->(Link),
(Link)-[:BATTLES]->(Ganondorf)
----
